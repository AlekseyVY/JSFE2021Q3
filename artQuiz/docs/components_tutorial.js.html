import Glide from '@glidejs/glide';
import View from '../core/view';
import state from '../state/state';

/**
 * Tutorial view class
 * @module Tutorial
 */
class Tutorial extends View {
  /**
   * @param {object} dto data transfer object
   */
  constructor(dto) {
    super(dto);
    this.dto = dto;
  }

  /**
   * renders Tutorila screen
   * @method render
   */
  render() {
    super.render();
    new Glide('.glide').mount();
  }
}
